CREATE USER LIBUSR PASSWORD "Password1" NO FORCE_FIRST_PASSWORD_CHANGE ;

CREATE ROLE Library_CROLE;

grant SELECT, UPDATE, INSERT, DELETE, EXECUTE, SELECT METADATA ON SCHEMA "LIBRARY" TO Library_CROLE with grant option;

grant Library_CROLE to LIBUSR with admin option;


SELECT *
FROM "SYS"."ROLES" WHERE ROLE_NAME = 'LIBRARY_CROLE';
